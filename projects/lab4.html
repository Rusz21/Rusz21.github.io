# Import Libraries 
import torch

from torch.utils.data import TensorDataset, DataLoader

1. DATA PREPARATION

In[2]:
# Generate 100 samples of data for a simple linear regression: Y = 2*X + 5 + noise
X_numpy = np.linspace(-5, 5, 100, dtype=np.float32).reshape(-1, 1) # 100 samples, 1 feature
Y_numpy = 2 * X_numpy + 5 + np.random.randn(*X_numpy.shape).astype(np.float32) * 0.5 # Target Y

# Convert numpy arrays to PyTorch Tensors
X_tensor = torch.from_numpy(X_numpy)
Y_tensor = torch.from_numpy(Y_numpy)

# Create a TensorDataset and DataLoader
dataset = TensorDataset(X_tensor, Y_tensor)
batch_size = 8 # Parameter from Task 4
data_loader = DataLoader(dataset, batch_size=batch_size, shuffle=True)

2. MODEL DEFINITION of Fully Connected Layers

In[3]:
# Assuming "Fully Connected Layers x 2" means a simple network with one hidden layer
# Input Feature: 1, Output Feature: 1
input_size = 1
hidden_size = 10 # Arbitrarily chosen for a simple hidden layer
output_size = 1

class SimpleNN(nn.Module):
    def __init__(self, input_size, hidden_size, output_size):
        super().__init__()
        # Fully Connected Layer 1 (Hidden Layer)
        self.fc1 = nn.Linear(input_size, hidden_size)
        # Activation (ReLU is common, though not specified, we'll use it)
        self.relu = nn.ReLU()
        # Fully Connected Layer 2 (Output Layer)
        self.fc2 = nn.Linear(hidden_size, output_size)

    def forward(self, x):
        out = self.fc1(x)
        out = self.relu(out)
        out = self.fc2(out)
        return out

model = SimpleNN(input_size, hidden_size, output_size)

3. LOSS, OPTIMIZER, AND TRAINING PARAMETERS

In[4]:
# Criterion: MSE Loss
criterion = nn.MSELoss() 

# Optimizer: SGD
# Learning rate is set low for stability
optimizer = torch.optim.SGD(model.parameters(), lr=0.01) 

# Epoch: 1000
num_epochs = 1000

4. FOR TRAINING LOOP

In[5]:
print("Starting Training...")
for epoch in range(num_epochs):
    for i, (inputs, targets) in enumerate(data_loader):
        
        # 1. Forward Pass
        outputs = model(inputs)
        loss = criterion(outputs, targets)
        
        # 2. Backward Pass (Zero gradients, Backpropagate, Update weights)
        optimizer.zero_grad() # Clear previous gradients
        loss.backward()       # Compute gradient of the loss w.r.t. model parameters
        optimizer.step()      # Update model parameters

    # Print loss every 100 epochs
    if (epoch + 1) % 100 == 0:
        print(f'Epoch [{epoch+1}/{num_epochs}], Loss: {loss.item():.4f}')

print("Training finished.")
